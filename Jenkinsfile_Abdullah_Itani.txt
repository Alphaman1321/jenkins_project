pipeline {
    agent any

    environment {
        VIRTUAL_ENV = "${env.WORKSPACE}\\venv"
        PATH = "C:\\Users\\user\\AppData\\Local\\Programs\\Python\\Python312\\;C:\\Users\\user\\AppData\\Local\\Programs\\Python\\Python312\\Scripts\\;${env.PATH}"
    }

    stages {
        stage('Setup') {
            steps {
                script {
                    if (!fileExists("${VIRTUAL_ENV}")) {
                        bat "python -m venv ${VIRTUAL_ENV}"
                    }
                    bat "${VIRTUAL_ENV}\\Scripts\\activate.bat && pip install -r requirements.txt"
                }
            }
        }

        stage('Lint') {
            steps {
                script {
                    bat "${VIRTUAL_ENV}\\Scripts\\activate.bat && flake8 app.py"
                }
            }
        }

        stage('Test') {
            steps {
                script {
                    bat "${VIRTUAL_ENV}\\Scripts\\activate.bat && set PYTHONPATH=${env.WORKSPACE} && pytest"
                }
            }
        }

        stage('Coverage') {
            steps {
                script {
                    bat "${VIRTUAL_ENV}\\Scripts\\activate.bat && coverage run -m pytest"
                    bat "${VIRTUAL_ENV}\\Scripts\\activate.bat && coverage report"
                    bat "${VIRTUAL_ENV}\\Scripts\\activate.bat && coverage html"  // Generates HTML report
                }
            }
        }

        stage('Security Scan') {
            steps {
                script {
                    bat "${VIRTUAL_ENV}\\Scripts\\activate.bat && bandit -r ."
                }
            }
        }

        stage('Deploy') {
            steps {
                script {
                    echo "Deploying application..."
                    // Placeholder for deployment logic
                    // Add commands here to deploy to a remote server, if applicable
                }
            }
        }
    }

    post {
        always {
            cleanWs()
        }
    }
}
